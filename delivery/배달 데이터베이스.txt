            * 배달 주문 어플 데이터베이스 * 
테이블
- 고객(사용자) : 이름, 이메일, 전화번호 
- 상점(가게) : 가게 이름, 위치, 연락처
- 주문 : 주문 번호, 주문한 사용자, 주문한 가게, 주문한 메뉴, 수량, 총 가격, 주문 시간
- 결제 : 결제 번호, 주문 번호, 결제 방법, 결제 금액, 결제 시간
- 메뉴 : 메뉴 이름, 설명, 가격

-- customer(고객) 테이블 생성
create table customer(
    user_id number primary key,
    name varchar2(20),
    email varchar2(50),
    phoneNumber varchar2(50)
);
-- shop(상점) 테이블 생성
create table shop(
    shop_id number primary key,
    name varchar2(50),
    location varchar2(100),
    shopNumber varchar2(50)
);
-- orders(주문) 테이블 생성
create table orders(
    order_id number primary key,
    user_id number,
    shop_id number,
    menu_id number,
    quantity number,
    price number,
    order_time varchar2(50),
    foreign key(user_id) references customer(user_id),
    foreign key(shop_id) references shop(shop_id),
    foreign key(menu_id) references menu(menu_id)
);
-- payment(결제) 테이블 생성
create table payment(
    payment_id number primary key,
    order_id number,
    paymentMethod varchar2(20),
    price number,
    foreign key(order_id) references orders(order_id)
);
-- menu(메뉴) 테이블 생성
create table menu(
    menu_id number primary key,
    name varchar2(50),
    description varchar2(200),
    price number
);
